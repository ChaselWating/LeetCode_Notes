```java
//利用双指针，但好像不太对
	public ListNode swapPairs0(ListNode head) {
		ListNode prev = head;
		ListNode cur = head.next;
		while(cur != null || prev != null) {
			prev.next = cur.next;
			cur.next = prev;
			prev = prev.next;
			cur = prev.next;
		}
    }
```
